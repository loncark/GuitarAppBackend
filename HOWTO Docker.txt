1. application.properties:

spring.jpa.show-sql=true
spring.datasource.url=jdbc:mysql://host.docker.internal:3306/testdb
spring.datasource.username=kristina
spring.datasource.password=password
spring.datasource.driver-class-name=com.mysql.cj.jdbc.Driver
spring.sql.init.mode=always

management.endpoints.web.exposure.include=*
management.endpoint.metrics.enabled=true
spring.main.allow-bean-definition-overriding=true

server.port=7000

2. Maven package + install

3. docker run -d -p 3306:3306 --name mysql -e MYSQL_PASSWORD=password -e MYSQL_DATABASE=testdb -e MYSQL_ROOT_PASSWORD=password mysql

4. docker build -t guitarapp .

5. docker run -d -p 7000:7000 guitarapp
